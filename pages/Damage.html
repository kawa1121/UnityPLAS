using UnityEngine;
using System.Collections;

public class Enemy : MonoBehaviour
{
    public int health = 100;
    // Stores the enemy's current HP（敵の現在の体力を保持）

    public Color damageColor = Color.red;
    // Color shown when taking damage（ダメージを受けたときの色）

    public float flashTime = 0.1f;
    // Duration of the damage effect（ダメージ演出の時間）

    private Renderer rend;
    // Component used to control appearance（見た目を操作するコンポーネント）

    private Color originalColor;
    // Saves the original color（元の色を保存）

    void Start()
    {
        rend = GetComponent();
        // Gets a component attached to this object（同じオブジェクトから取得）

        originalColor = rend.material.color;
        // Stores the starting color（初期の色を記録）
    }

    void OnCollisionEnter(Collision collision)
    {
        // Called automatically when a collision happens（衝突時に自動で呼ばれる）

        if (collision.gameObject.CompareTag("Bullet"))
        {
            // Checks the tag of the collided object（衝突相手のタグを確認）

            Destroy(collision.gameObject);
            // Removes the collided object from the scene（衝突したオブジェクトを削除）

            health -= 20;
            // Decreases the HP value（体力を減らす）

            Debug.Log("Enemy HP: " + health);
            // Outputs current HP to the console（現在の体力を表示）

            StartCoroutine(DamageFlash());
            // Starts a damage effect over time（時間を使った演出を開始）

            if (health <= 0)
            {
                // Checks if the enemy should be destroyed（敵が倒されたか判定）

                Destroy(gameObject);
                // Removes this enemy from the scene（敵自身を削除）
            }
        }
    }

    IEnumerator DamageFlash()
    {
        rend.material.color = damageColor;
        // Changes color to show damage（ダメージ色に変更）

        yield return new WaitForSeconds(flashTime);
        // Waits for a short time（一定時間待つ）

        rend.material.color = originalColor;
        // Restores the original color（元の色に戻す）
    }
}

This script handles the enemy’s damage processing when it collides with a bullet.
When the enemy collides with an object tagged as "Bullet", the bullet is destroyed and the enemy’s health is reduced by a fixed amount. The current health value is printed to the console for debugging purposes.
At the same time, a visual damage effect is played. The enemy temporarily changes its color to indicate that it has been hit, then returns to its original color after a short delay using a coroutine.
If the enemy’s health reaches zero or below, the enemy GameObject is destroyed and removed from the scene.
This script combines collision detection, health management, visual feedback, and object destruction to implement a basic enemy damage system.

(The Renderer component is responsible for drawing an object on the screen.
It controls how the object looks, such as its color, material, and texture.
By changing the material’s color through the Renderer, you can create visual effects like damage flashes or color changes when the object is hit.)
