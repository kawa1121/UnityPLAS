using UnityEngine;

public class PlayerFriction : MonoBehaviour
{
    public float friction = 5f;
   // Controls how quickly movement slows down（減速の速さを制御）
    private Vector3 velocity;
    // Stores current movement speed and direction（現在の移動速度と方向）
    void Update()
    {
        float h = Input.GetAxis("Horizontal");
        float v = Input.GetAxis("Vertical");
        Vector3 input = new Vector3(h, 0, v);

        velocity += input;
        // Accumulates movement over time（移動量を蓄積） 
        if (velocity.magnitude > 0f)
        // Checks the length of the velocity vector（速度ベクトルの大きさを確認） 
        {
            velocity = Vector3.Lerp(velocity, Vector3.zero, friction * Time.deltaTime);
        // Gradually moves a value toward zero（値を徐々に0へ近づける）    
        // Uses frame time for smooth change（フレーム間時間を使用）
        }

        transform.Translate(velocity * Time.deltaTime, Space.World);
        // Moves the object using current velocity（現在の速度を使って移動）
    }
}
