using UnityEngine;
// Unity features can be used（Unity の機能を使えるようにする）

public class PlayerMovement : MonoBehaviour
// Allows this script to be attached to a GameObject（GameObject にアタッチ可能にする）
{
    public float speed = 5f;

    void Update()
    {
        // Called every frame by Unity（Unity が毎フレーム呼び出す）

        float h = Input.GetAxis("Horizontal");
        // Gets player input（プレイヤー入力を取得）

        float v = Input.GetAxis("Vertical");
        // Gets player input（プレイヤー入力を取得）

        Vector3 direction = new Vector3(h, 0, v);
        // 3D direction or position（3D の方向・位置）

        transform.Translate(
            direction * speed * Time.deltaTime,
            Space.World
        );
        // Moves the GameObject（GameObject を移動させる）
        // Frame-rate independent timing（フレームレートに依存しない時間）
        // Uses world coordinates（ワールド座標を使用）
    }
}
