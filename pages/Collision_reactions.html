using UnityEngine;

public class BounceOnCollision : MonoBehaviour
{
    public float bounceUpForce = 6f;
    // Initial upward bounce strength（最初の跳ね上がる強さ）

    public float damping = 0.7f;
    // Reduces bounce strength after each collision（衝突ごとに跳ね返りを弱くする）

    public float stopThreshold = 0.3f;
    // Threshold to stop bouncing completely（完全停止するためのしきい値）

    private Rigidbody rb;
    // Physics component used for movement（物理挙動を扱うコンポーネント）

    private float currentBounceForce;
    // Current bounce strength that changes over time（時間とともに変化する現在の跳ね返り力）

    void Start()
    {
        rb = GetComponent();
        // Gets a physics component from the same object（同じオブジェクトから取得）

        currentBounceForce = bounceUpForce;
        // Initializes the current bounce force（現在の跳ね返り力を初期化）

        rb.velocity = new Vector3(4f, 0f, 0f);
        // Sets an initial horizontal velocity（初期の横方向速度を設定）
    }

    void OnCollisionEnter(Collision collision)
    {
        // Called automatically when a collision occurs（衝突時に自動で呼ばれる）

        Vector3 velocity = rb.velocity;
        // Stores the current velocity（現在の速度を保持）

        float horizontalSpeedX = velocity.x;
        float horizontalSpeedZ = velocity.z;
        // Preserves horizontal movement（横方向の動きを保持）

        if (currentBounceForce < stopThreshold)
        {
            // Checks if bouncing should stop（跳ね返りを止めるか判定）

            rb.velocity = Vector3.zero;
            rb.isKinematic = true;
            // Stops physics-based movement（物理挙動を停止させる)
            return;
        }

        velocity.x = horizontalSpeedX;
        velocity.z = horizontalSpeedZ;
        velocity.y = currentBounceForce;
        // Applies upward velocity only（縦方向のみ跳ね返りを与える）

        rb.velocity = velocity;
        // Overwrites the Rigidbody velocity（速度を直接設定）

        currentBounceForce *= damping;
        // Weakens the next bounce（次のバウンドを弱くする）
    }
}

This script makes a GameObject bounce upward every time it collides with another object.
The bounce strength gradually decreases after each collision, and the object eventually stops moving.
The object keeps its horizontal movement while bouncing, creating a realistic sliding and bouncing motion.
Once the bounce strength becomes smaller than a specified threshold, physics simulation is disabled.
