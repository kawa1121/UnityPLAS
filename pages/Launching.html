using UnityEngine;

public class Shooting : MonoBehaviour
{
    public GameObject bulletPrefab;
    // Basic unit of objects in a scene（シーン内オブジェクトの基本単位）

    public Transform firePoint;
    // Transform used as a launch position（発射位置として使われる Transform）

    public float bulletSpeed = 10f;
    // Controls how fast the bullet moves（弾の速さを制御）

    void Update()
    {
        if (Input.GetButtonDown("Fire1"))
        {
            // Detects the moment the fire button is pressed（発射ボタンが押された瞬間を検出）

            GameObject bullet = Instantiate(
                bulletPrefab,
                firePoint.position,
                firePoint.rotation
            );
            // Creates a new object in the scene（新しいオブジェクトを生成）
            // Uses Transform position information（Transform の位置情報を使用）

            Rigidbody rb = bullet.GetComponent();
            // Gets a physics component from the object（物理コンポーネントを取得）

            rb.velocity = firePoint.forward * bulletSpeed;
            // Forward direction vector of the object（オブジェクトの正面方向ベクトル）
            // Directly sets movement speed and direction（移動速度と方向を直接設定）
        }
    }
}
